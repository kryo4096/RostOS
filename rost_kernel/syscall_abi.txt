HOW TO SYSCALL

mov rdx, {syscall type}
int 80h


0h - print
DESC
    prints an ASCII message to the VGA console

ARGS
    rsi = message pointer (u64)
    rdx = character count (u64)

RETURNS
    (none)


----

1h - println
DESC
    prints an ASCII message + '\n' to the VGA console

ARGS
    rsi = message pointer (u64)
    rdx = character count (u64)

RETURNS
    (none)

----

2h - debug
DESC
    prints a number to the VGA console in the specified format

ARGS
    rsi = number to print (u64)
    rdx = format option
        0x0 binary
        0x1 octal
        0x2 decimal
        0x3 hexadecimal
RETURNS
    (none)

----

10h - time 
DESC
    returns the number of ticks passed since kernel startup
ARGS
    (none)

RETURNS
    rax = ticks (u64)

----


20h - umx (User Memory eXtend) 
DESC
    extends user memory by the specified amount of pages

ARGS
    rsi = amount of pages to add (u64)

RETURNS
    rax = new memory end address
